#include <stdio.h>
#include <string.h>
#include <stdbool.h>

// تحديد حجم المخزن
#define SIZE 20 

// تعريف هيكل المخزن الدائري
typedef struct {
    char buffer[SIZE]; // مصفوفة تخزيين البيانات
    int head;          // مؤشر القراءة
    int tail;          // مؤشر الكتابة
    int count;         // عدد العناصر الحالية
} CircularBuffer;

// دالة تهيئة المخزن
void init(CircularBuffer *cb) {
    cb->head = 0;
    cb->tail = 0;
    cb->count = 0;
}

// فحص حالة المخزن عند الامتلاء
bool isFull(CircularBuffer *cb) {
    return cb->count == SIZE;
}

// فحص حالة المخزن الفارغ
bool isEmpty(CircularBuffer *cb) {
    return cb->count == 0;
}

// دالة الكتابة
void write(CircularBuffer *cb, char data) {
    if (isFull(cb)) {
        printf("\n Error OverFlow '%c'\n", data);
        return;
    }
    cb->buffer[cb->tail] = data;
    cb->tail = (cb->tail + 1) % SIZE; 
    cb->count++;
}

// دالة القراءة
char read(CircularBuffer *cb) {
    if (isEmpty(cb)) {
        printf("\n Error UnderFlow\n");
        return '\0';
    }
    char data = cb->buffer[cb->head];
    cb->head = (cb->head + 1) % SIZE;
    cb->count--;
    return data;
}

int main() {
    CircularBuffer cb;
    init(&cb); // تهيئة المخزن عند البداية

    char name[50];
    const char *suffix = "CE-ESY";

    // 1. طلب الاسم من المستخدم
    printf("Enter your name: ");
    scanf("%s", name);
    strcat(name, suffix);
    printf("name after change: %s\n", name);
    // 3. تخزين النص في المخزن الدائري حرفاً بحرف
    printf("writing in the storage...\n");
    for (int i = 0; i < strlen(name); i++) {
        write(&cb, name[i]);
    }

    // 4. القراءة من المخزن وعرض النتيجة
    printf("resualt from the stor: ");
    while (!isEmpty(&cb)) {
        printf("%c", read(&cb));
    }
    printf("\n");

    // 5. التأكد من حالة المخزن النهائية
    if (isEmpty(&cb)) {
        printf("confirm: the store is empty now \n");
    }

    return 0;
}
